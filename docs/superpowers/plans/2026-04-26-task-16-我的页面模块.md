- [x] **Step 4: 创建 personal_center_provider.dart**

```dart
import 'package:flutter_riverpod/flutter_riverpod.dart';
import '../../domain/personal_center_repository.dart';
import '../../domain/user_info.dart';
import '../../domain/entry_item.dart';
import '../../data/mock_personal_center_repository.dart';

final personalCenterRepositoryProvider = Provider<PersonalCenterRepository>((ref) {
  return MockPersonalCenterRepository();
});

final userInfoProvider = FutureProvider<UserInfo>((ref) async {
  final repository = ref.watch(personalCenterRepositoryProvider);
  return repository.getUserInfo();
});

final businessEntriesProvider = Provider<List<EntryItem>>((ref) {
  final repository = ref.watch(personalCenterRepositoryProvider);
  return repository.getBusinessEntries();
});

final systemEntriesProvider = Provider<List<EntryItem>>((ref) {
  final repository = ref.watch(personalCenterRepositoryProvider);
  return repository.getSystemEntries();
});
```

- [x] **Step 5: 创建 personal_center_page.dart**

```dart
import 'package:flutter/material.dart';
import 'package:flutter_riverpod/flutter_riverpod.dart';
import 'package:go_router/go_router.dart';
import '../../../../core/theme/app_colors.dart';
import '../providers/personal_center_provider.dart';

class PersonalCenterPage extends ConsumerWidget {
  const PersonalCenterPage({super.key});

  void _showUpdateDialog(BuildContext context) {
    showDialog(
      context: context,
      builder: (context) => AlertDialog(
        title: const Text('检查更新'),
        content: const Text('当前已是最新版本'),
        actions: [
          TextButton(
            onPressed: () => Navigator.pop(context),
            child: const Text('确定'),
          ),
        ],
      ),
    );
  }

  void _showAboutDialog(BuildContext context) {
    showDialog(
      context: context,
      builder: (context) => AboutDialog(
        applicationName: '奶牛药浴大数据管理中心',
        applicationVersion: '1.0.0',
        applicationIcon: Container(
          width: 48,
          height: 48,
          decoration: BoxDecoration(
            color: AppColors.primaryOrange,
            borderRadius: BorderRadius.circular(8),
          ),
          child: const Icon(Icons.agriculture, color: Colors.white),
        ),
        children: const [
          Text('Scaling Robotics'),
          Text('提供奶牛药浴大数据管理服务'),
        ],
      ),
    );
  }

  IconData _getIcon(String iconName) {
    switch (iconName) {
      case 'security':
        return Icons.security;
      case 'assignment_turned_in':
        return Icons.assignment_turned_in;
      case 'build':
        return Icons.build;
      case 'settings':
        return Icons.settings;
      case 'system_update':
        return Icons.system_update;
      case 'info':
        return Icons.info;
      default:
        return Icons.arrow_forward_ios;
    }
  }

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final userInfoAsync = ref.watch(userInfoProvider);
    final businessEntries = ref.watch(businessEntriesProvider);
    final systemEntries = ref.watch(systemEntriesProvider);

    return Scaffold(
      backgroundColor: AppColors.backgroundGray,
      appBar: AppBar(
        title: const Text('个人中心'),
      ),
      body: ListView(
        children: [
          // 用户信息区
          userInfoAsync.when(
            loading: () => const SizedBox(height: 80),
            error: (_, __) => const SizedBox(height: 80),
            data: (userInfo) => Container(
              color: Colors.white,
              padding: const EdgeInsets.all(16),
              child: Row(
                children: [
                  Expanded(
                    child: Text(
                      userInfo.displayText,
                      style: const TextStyle(
                        fontSize: 16,
                        color: AppColors.textPrimary,
                      ),
                    ),
                  ),
                  const Text('个人信息'),
                  const Icon(Icons.arrow_forward_ios, size: 16),
                ],
              ),
            ),
          ),

          const SizedBox(height: 12),

          // 业务相关入口
          Card(
            margin: const EdgeInsets.symmetric(horizontal: 12, vertical: 6),
            child: Column(
              children: businessEntries.asMap().entries.map((entry) {
                final index = entry.key;
                final item = entry.value;
                return Column(
                  children: [
                    ListTile(
                      leading: Icon(
                        _getIcon(item.icon),
                        color: AppColors.primaryOrange,
                      ),
                      title: Text(item.name),
                      trailing: const Icon(Icons.arrow_forward_ios, size: 16),
                      onTap: () {
                        switch (item.name) {
                          case '维保计划':
                            context.push('/maintenance-plan');
                            break;
                          case '维保记录':
                            context.push('/maintenance-record');
                            break;
                          case '故障日志':
                            context.push('/fault-log');
                            break;
                        }
                      },
                    ),
                    if (index < businessEntries.length - 1)
                      const Divider(height: 1, indent: 56),
                  ],
                );
              }).toList(),
            ),
          ),

          // 系统相关入口
          Card(
            margin: const EdgeInsets.symmetric(horizontal: 12, vertical: 6),
            child: Column(
              children: systemEntries.asMap().entries.map((entry) {
                final index = entry.key;
                final item = entry.value;
                return Column(
                  children: [
                    ListTile(
                      leading: Icon(
                        _getIcon(item.icon),
                        color: AppColors.textSecondary,
                      ),
                      title: Text(item.name),
                      trailing: const Icon(Icons.arrow_forward_ios, size: 16),
                      onTap: () {
                        if (item.name == '系统设置') {
                          context.push('/settings');
                        } else if (item.name == '检查更新') {
                          // TODO: 检查更新功能（显示版本信息或弹窗提示）
                          _showUpdateDialog(context);
                        } else if (item.name == '关于我们') {
                          // TODO: 关于我们功能（显示应用信息和公司信息）
                          _showAboutDialog(context);
                        }
                      },
                    ),
                    if (index < systemEntries.length - 1)
                      const Divider(height: 1, indent: 56),
                  ],
                );
              }).toList(),
            ),
          ),

          const SizedBox(height: 24),

          // 退出登录按钮
          Padding(
            padding: const EdgeInsets.symmetric(horizontal: 12),
            child: OutlinedButton(
              onPressed: () {
                // TODO: 退出登录逻辑
                context.go('/login');
              },
              style: OutlinedButton.styleFrom(
                foregroundColor: AppColors.primaryOrange,
                side: const BorderSide(color: AppColors.primaryOrange),
                padding: const EdgeInsets.symmetric(vertical: 16),
              ),
              child: const Text('退出登录'),
            ),
          ),

          const SizedBox(height: 24),
        ],
      ),
    );
  }
}
```
