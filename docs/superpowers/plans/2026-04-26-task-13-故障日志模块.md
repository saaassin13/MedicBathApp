```dart
/// 故障日志级别
enum FaultLevel {
  error,   // 错误 - 红色
  warning, // 警告 - 橙色
  info,    // 提示 - 绿色
}

/// 故障日志项
class FaultLogItem {
  final String timestamp; // MM/dd HH:mm:ss.SSS
  final FaultLevel level;
  final String description;

  const FaultLogItem({
    required this.timestamp,
    required this.level,
    required this.description,
  });

  String get levelLabel {
    switch (level) {
      case FaultLevel.error:
        return '错误';
      case FaultLevel.warning:
        return '警告';
      case FaultLevel.info:
        return '提示';
    }
  }
}
```

- [x] **Step 4: 创建 fault_log_provider.dart**

```dart
import 'package:flutter_riverpod/flutter_riverpod.dart';
import '../../domain/fault_log_repository.dart';
import '../../domain/fault_log_item.dart';
import '../../data/mock_fault_log_repository.dart';

final faultLogRepositoryProvider = Provider<FaultLogRepository>((ref) {
  return MockFaultLogRepository();
});

final faultLogsProvider = FutureProvider<List<FaultLogItem>>((ref) async {
  final repository = ref.watch(faultLogRepositoryProvider);
  return repository.getFaultLogs();
});
```

- [x] **Step 5: 创建 fault_log_page.dart**

```dart
import 'package:flutter/material.dart';
import 'package:flutter_riverpod/flutter_riverpod.dart';
import 'package:go_router/go_router.dart';
import '../../../../core/theme/app_colors.dart';
import '../providers/fault_log_provider.dart';
import '../../domain/fault_log_item.dart';

class FaultLogPage extends ConsumerWidget {
  const FaultLogPage({super.key});

  Color _getLevelColor(FaultLevel level) {
    switch (level) {
      case FaultLevel.error:
        return AppColors.errorRed;
      case FaultLevel.warning:
        return AppColors.warningOrange;
      case FaultLevel.info:
        return AppColors.successGreen;
    }
  }

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final logsAsync = ref.watch(faultLogsProvider);

    return Scaffold(
      backgroundColor: AppColors.backgroundGray,
      appBar: AppBar(
        leading: IconButton(
          icon: const Icon(Icons.close),
          onPressed: () => context.go('/personal-center'),
        ),
        title: const Text('双城牧场-一期奶厅-一号转盘'),
      ),
      body: logsAsync.when(
        loading: () => const Center(child: CircularProgressIndicator()),
        error: (error, stack) => Center(child: Text('加载失败: $error')),
        data: (logs) => ListView(
          children: [
            Card(
              margin: const EdgeInsets.all(12),
              child: Column(
                crossAxisAlignment: CrossAxisAlignment.start,
                children: [
                  const Padding(
                    padding: EdgeInsets.all(12),
                    child: Row(
                      children: [
                        Icon(Icons.build, color: AppColors.primaryOrange),
                        SizedBox(width: 8),
                        Text(
                          '故障日志查询',
                          style: TextStyle(fontWeight: FontWeight.bold),
                        ),
                      ],
                    ),
                  ),
                  const Divider(height: 1),
                  ...logs.map((log) => Padding(
                        padding: const EdgeInsets.symmetric(
                          horizontal: 12,
                          vertical: 8,
                        ),
                        child: Row(
                          children: [
                            Text(
                              log.timestamp,
                              style: const TextStyle(
                                color: AppColors.textSecondary,
                                fontSize: 12,
                              ),
                            ),
                            const SizedBox(width: 8),
                            Container(
                              padding: const EdgeInsets.symmetric(
                                horizontal: 6,
                                vertical: 2,
                              ),
                              decoration: BoxDecoration(
                                color: _getLevelColor(log.level),
                                borderRadius: BorderRadius.circular(4),
                              ),
                              child: Text(
                                '[${log.levelLabel}]',
                                style: const TextStyle(
                                  color: Colors.white,
                                  fontSize: 10,
                                ),
                              ),
                            ),
                            const SizedBox(width: 8),
                            Expanded(
                              child: Text(
                                log.description,
                                style: const TextStyle(
                                  color: AppColors.textPrimary,
                                ),
                              ),
                            ),
                          ],
                        ),
                      )),
                ],
              ),
            ),
            const Center(
              child: Padding(
                padding: EdgeInsets.all(16),
                child: Text(
                  '点击或下拉加载更多...',
                  style: TextStyle(color: AppColors.textSecondary),
                ),
              ),
            ),
          ],
        ),
      ),
    );
  }
}
```
