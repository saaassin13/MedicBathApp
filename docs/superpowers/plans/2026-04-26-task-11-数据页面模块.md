```dart
import 'package:flutter/material.dart';
import 'package:fl_chart/fl_chart.dart';
import '../../../../core/theme/app_colors.dart';
import '../../domain/entities.dart';

class ChartCard extends StatelessWidget {
  final String title;
  final ChartData data;

  const ChartCard({super.key, required this.title, required this.data});

  @override
  Widget build(BuildContext context) {
    return Card(
      child: Padding(
        padding: const EdgeInsets.all(12),
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            Text(title, style: const TextStyle(fontWeight: FontWeight.bold)),
            const SizedBox(height: 16),
            SizedBox(
              height: 200,
              child: _buildCombinedChart(),
            ),
          ],
        ),
      ),
    );
  }

  Widget _buildCombinedChart() {
    final maxCowCount = data.cowCounts.reduce((a, b) => a > b ? a : b);
    final maxYLeft = (maxCowCount * 1.2).toDouble();
    const maxYRight = 100.0;

    return BarChart(
      BarChartData(
        alignment: BarChartAlignment.spaceAround,
        maxY: maxYLeft,
        barGroups: data.cowCounts.asMap().entries.map((entry) {
          return BarChartGroupData(
            x: entry.key,
            barRods: [
              // 蓝色柱 - 药浴牛数（左Y轴）
              BarChartRodData(
                toY: entry.value.toDouble(),
                color: AppColors.chartBlue,
                width: 12,
                borderRadius: const BorderRadius.only(
                  topLeft: Radius.circular(2),
                  topRight: Radius.circular(2),
                ),
              ),
            ],
          );
        }).toList(),
        titlesData: FlTitlesData(
          show: true,
          bottomTitles: AxisTitles(
            sideTitles: SideTitles(
              showTitles: true,
              reservedSize: 30,
              getTitles: (value) {
                final index = value.toInt();
                if (index >= 0 && index < data.dates.length) {
                  return data.dates[index];
                }
                return '';
              },
            ),
          ),
          leftTitles: AxisTitles(
            axisNameWidget: const Text('牛数', style: TextStyle(fontSize: 10)),
            sideTitles: SideTitles(
              showTitles: true,
              reservedSize: 40,
              interval: maxYLeft / 4,
              getTitles: (value) => value.toInt().toString(),
            ),
          ),
          rightTitles: AxisTitles(
            axisNameWidget: const Text('识别率', style: TextStyle(fontSize: 10)),
            sideTitles: SideTitles(
              showTitles: true,
              reservedSize: 40,
              interval: 20,
              getTitles: (value) => '${value.toInt()}%',
            ),
          ),
          topTitles: const AxisTitles(sideTitles: SideTitles(showTitles: false)),
        ),
        borderData: FlBorderData(show: false),
        gridData: FlGridData(
          show: true,
          drawVerticalLine: false,
          horizontalInterval: maxYLeft / 4,
          getDrawingHorizontalLine: (value) => FlLine(
            color: Colors.grey[300]!,
            strokeWidth: 1,
          ),
        ),
      ),
    );
  }
}

/// 折线图数据卡片（独立组件，用于在数据页面显示乳头识别率折线）
class RecognitionRateLineChart extends StatelessWidget {
  final ChartData data;

  const RecognitionRateLineChart({super.key, required this.data});

  @override
  Widget build(BuildContext context) {
    return LineChart(
      LineChartData(
        gridData: const FlGridData(show: false),
        titlesData: const FlTitlesData(show: false),
        borderData: FlBorderData(show: false),
        lineBarsData: [
          LineChartBarData(
            spots: data.recognitionRates.asMap().entries.map((e) {
              // 将百分比（99.5-99.9）缩放到合适范围用于显示
              return FlSpot(e.key.toDouble(), e.value);
            }).toList(),
            isCurved: true,
            color: AppColors.primaryOrange,
            barWidth: 2,
            dotData: const FlDotData(show: true),
            belowBarData: BarAreaData(show: false),
          ),
        ],
        minY: 99,
        maxY: 100,
      ),
    );
  }
}
```

> **注意**：由于 fl_chart 限制，复合图采用 LineChart 配合 barWidth 实现柱状效果。柱状图（蓝色，药浴牛数）对应左Y轴；折线图（橙色，乳头识别率）对应右Y轴，使用 LineChart 的双轴特性实现。具体实现时需要额外配置 LineChartBarData 的 barWidth 属性来模拟柱状效果，或使用 fl_chart 0.65+ 的 CombinedChart（如果可用）。
