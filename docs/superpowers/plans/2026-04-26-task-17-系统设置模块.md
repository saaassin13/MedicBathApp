```dart
import '../../domain/settings_repository.dart';

class MockSettingsRepository implements SettingsRepository {
  bool _alarmPushEnabled = true;

  @override
  Future<bool> getAlarmPushEnabled() async {
    await Future.delayed(const Duration(milliseconds: 200));
    return _alarmPushEnabled;
  }

  @override
  Future<void> setAlarmPushEnabled(bool enabled) async {
    await Future.delayed(const Duration(milliseconds: 100));
    _alarmPushEnabled = enabled;
  }

  @override
  Future<void> clearCache() async {
    await Future.delayed(const Duration(milliseconds: 500));
  }
}
```

- [x] **Step 4: 创建 settings_provider.dart**

```dart
import 'package:flutter_riverpod/flutter_riverpod.dart';
import '../../domain/settings_repository.dart';
import '../../domain/setting_item.dart';
import '../../data/mock_settings_repository.dart';

final settingsRepositoryProvider = Provider<SettingsRepository>((ref) {
  return MockSettingsRepository();
});

final alarmPushEnabledProvider = FutureProvider<bool>((ref) async {
  final repository = ref.watch(settingsRepositoryProvider);
  return repository.getAlarmPushEnabled();
});

class AlarmPushNotifier extends StateNotifier<bool> {
  final SettingsRepository _repository;

  AlarmPushNotifier(this._repository) : super(true);

  Future<void> toggle() async {
    final newValue = !state;
    await _repository.setAlarmPushEnabled(newValue);
    state = newValue;
  }
}

final alarmPushNotifierProvider =
    StateNotifierProvider<AlarmPushNotifier, bool>((ref) {
  final repository = ref.watch(settingsRepositoryProvider);
  return AlarmPushNotifier(repository);
});
```

- [x] **Step 5: 创建 settings_page.dart**

```dart
import 'package:flutter/material.dart';
import 'package:flutter_riverpod/flutter_riverpod.dart';
import 'package:go_router/go_router.dart';
import '../../../../core/theme/app_colors.dart';
import '../providers/settings_provider.dart';

class SettingsPage extends ConsumerWidget {
  const SettingsPage({super.key});

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final alarmPushEnabled = ref.watch(alarmPushNotifierProvider);

    return Scaffold(
      backgroundColor: AppColors.backgroundGray,
      appBar: AppBar(
        leading: IconButton(
          icon: const Icon(Icons.close),
          onPressed: () => context.pop(),
        ),
        title: const Text('系统设置'),
      ),
      body: Card(
        margin: const EdgeInsets.all(12),
        child: Column(
          children: [
            // 异常告警推送
            ListTile(
              leading: Container(
                padding: const EdgeInsets.all(6),
                decoration: BoxDecoration(
                  color: AppColors.warningOrange.withOpacity(0.1),
                  borderRadius: BorderRadius.circular(4),
                ),
                child: const Icon(
                  Icons.warning,
                  color: AppColors.warningOrange,
                  size: 20,
                ),
              ),
              title: const Text('异常告警推送'),
              trailing: Switch(
                value: alarmPushEnabled,
                onChanged: (value) {
                  ref.read(alarmPushNotifierProvider.notifier).toggle();
                },
                activeColor: AppColors.successGreen,
              ),
            ),
            const Divider(height: 1),
            // 清除缓存
            ListTile(
              leading: Container(
                padding: const EdgeInsets.all(6),
                decoration: BoxDecoration(
                  color: AppColors.textSecondary.withOpacity(0.1),
                  borderRadius: BorderRadius.circular(4),
                ),
                child: const Icon(
                  Icons.delete_outline,
                  color: AppColors.textSecondary,
                  size: 20,
                ),
              ),
              title: const Text('清除缓存'),
              onTap: () {
                showDialog(
                  context: context,
                  builder: (context) => AlertDialog(
                    title: const Text('清除缓存'),
                    content: const Text('确定要清除缓存吗？'),
                    actions: [
                      TextButton(
                        onPressed: () => Navigator.pop(context),
                        child: const Text('取消'),
                      ),
                      TextButton(
                        onPressed: () async {
                          Navigator.pop(context);
                          final repository = ref.read(settingsRepositoryProvider);
                          await repository.clearCache();
                          if (context.mounted) {
                            ScaffoldMessenger.of(context).showSnackBar(
                              const SnackBar(content: Text('缓存已清除')),
                            );
                          }
                        },
                        child: const Text('确定'),
                      ),
                    ],
                  ),
                );
              },
            ),
          ],
        ),
      ),
    );
  }
}
```
